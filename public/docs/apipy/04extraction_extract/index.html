def extract_frame_ffmpeg(video_path, cap, frame_number) -> Optional[np.ndarray]:
    """
    Utilizza ffmpeg per estrarre il frame specificato dal video.
    Se il frame non esiste, restituisce None.

    Args:
        video_path (str): Percorso del video da cui estrarre il frame
        frame_number (int): Numero del frame da estrarre
        cap (cv2.VideoCapture): Oggetto OpenCV VideoCapture per il video specificato

    Returns:
        (Optional[np.ndarray]): Un array BGR rappresentante l'immagine estratta, o None se il frame non esiste.
    """
    # 1. Ottieni fps per calcolare il timestamp
    fps = cap.get(cv2.CAP_PROP_FPS)

    # Calcola il timestamp (in secondi)
    timestamp = frame_number / fps

    # 2. Costruisci il comando ffmpeg
    cmd = [
        'ffmpeg',
        '-ss', f'{timestamp:.3f}',   # seek al timestamp
        '-i', video_path,            # video di input
        '-frames:v', '1',            # estrai un solo frame
        '-f', 'image2pipe',          # formato pipe
        '-vcodec', 'png',            # codec di output
        '-'                          # stdout
    ]

    # 3. Esegui il comando e leggi i byte PNG da stdout
    proc = subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.DEVNULL)
    raw_image = proc.stdout.read()
    proc.stdout.close()
    proc.wait()

    if not raw_image:
        print(f"Errore: Frame {frame_number} non trovato in {video_path} (ffmpeg)")
        return None

    # 4. Decodifica il PNG in un array BGR con OpenCV
    arr = np.frombuffer(raw_image, np.uint8)
    frame = cv2.imdecode(arr, cv2.IMREAD_COLOR)
    if frame is None:
        print(f"Errore: impossibile decodificare il frame {frame_number} di {video_path}")
        return None
    return frame

def extract_frames_from_video(video_path, video_frames_dir):
    """
    Utilizza ffmpeg per estrarre i frame dal video. Se il video non esiste, restituisce False.
    Se si verifica un errore durante l'estrazione, restituisce False.

    Args:
        video_path (str): Percorso del video da elaborare
        video_frames_dir (str): Percorso della cartella dove salvare i frame

    Returns:
        (bool): True se l'estrazione va a buon fine, False altrimenti
    """
    try:
        subprocess.run([
            "ffmpeg",
            "-i", video_path,         # input
            "-q:v", "2",              # qualità JPEG (1-31, più piccolo = migliore)
            os.path.join(video_frames_dir, "frame_%05d.jpg")
        ])
        return True

    except Exception as e:
        print(f"Errore: {e}")
        return False

def extract_keypoints_from_frame(frame, frame_number, fps, pose):    
    """
    Utilizza MediaPipe per estrarre i keypoints dal frame specificato.
    Se il frame non esiste, restituisce None.

    Args:
        frame (numpy.ndarray): Array BGR rappresentante l'immagine estratta
        frame_number (int): Numero del frame estratto
        fps (float): Numero di frame al secondo
        pose (mp_pose.Pose): Oggetto MediaPipe per l'estrazione dei keypoints

    Returns:
        (dict): Dizionario contenente i keypoints estratti
    """
    height, width, _ = frame.shape
    results = pose.process(cv2.cvtColor(frame, cv2.COLOR_BGR2RGB))

    # Inizializza dizionario con 0
    keypoints = {
        # --- metadati temporali ---
        'frame': frame_number,
        'timestamp': round(frame_number / fps, 3) if fps else None,
        'fps': fps,
        'mean_visibility': 0.0,

        # --- keypoints 2D ---
        'nose_x': 0, 'nose_y': 0,
        'neck_x': 0, 'neck_y': 0,
        'RShoulder_x': 0, 'RShoulder_y': 0,
        'RElbow_x': 0, 'RElbow_y': 0,
        'RWrist_x': 0, 'RWrist_y': 0,
        'LShoulder_x': 0, 'LShoulder_y': 0,
        'LElbow_x': 0, 'LElbow_y': 0,
        'LWrist_x': 0, 'LWrist_y': 0,
        'RHip_x': 0, 'RHip_y': 0,
        'RKnee_x': 0, 'RKnee_y': 0,
        'RAnkle_x': 0, 'RAnkle_y': 0,
        'LHip_x': 0, 'LHip_y': 0,
        'LKnee_x': 0, 'LKnee_y': 0,
        'LAnkle_x': 0, 'LAnkle_y': 0,
        'REye_x': 0, 'REye_y': 0,
        'LEye_x': 0, 'LEye_y': 0,
        'REar_x': 0, 'REar_y': 0,
        'LEar_x': 0, 'LEar_y': 0
    }

    if results.pose_landmarks:
        lm = results.pose_landmarks.landmark

        # Calcola la visibilità media (confidence globale)
        visibilities = [l.visibility for l in lm]
        keypoints['mean_visibility'] = sum(visibilities) / len(visibilities)

        for name, idx in mp_index.items():
            if idx is not None:
                x = int(lm[idx].x * width)
                y = int(lm[idx].y * height)
                keypoints[f"{name}_x"] = x
                keypoints[f"{name}_y"] = y

        # Calcolo del collo come punto medio tra le spalle (opzionale)
        if results.pose_landmarks:
            try:
                l_sh = lm[11]
                r_sh = lm[12]
                neck_x = int(((l_sh.x + r_sh.x) / 2) * width)
                neck_y = int(((l_sh.y + r_sh.y) / 2) * height)
                keypoints['neck_x'] = neck_x
                keypoints['neck_y'] = neck_y
            except:
                pass

    return keypoints

def kp_extraction(video_frames_dir, video_path, output_temp_csv):
    """
    Estrae i keypoints dai frame di un video utilizzando Mediapipe e li salva in un file JSON.

    Args: 
        video_frames_dir (str): Percorso della cartella contenente i frame
        video_path (str): Percorso del video da elaborare
        output_temp_csv (str): Percorso del file JSON dove salvare i keypoints

    Returns:     
        (tuple):
            (success: bool, code: 0) OK,
            (success: bool, code: 1) Errore generico
            (success: bool, code: 2) Video non trovato
            (success: bool, code: 3) Troppi frame illeggibili
            (success: bool, code: 4) File non trovato
    """
    BATCH_SIZE = 10
    MAX_TOTAL_ERRORS_PERCENTAGE = 0.1 # 10% dei frame totali
    MAX_GAP_SECONDS = 0.5 # percentuale in secondi del video   

    buffer = []
    total_frame_error_count = 0
    consecutive_frame_error_count = 0

    # VIDEO SINGOLO
    cap = cv2.VideoCapture(video_path)
    if not cap.isOpened():
        return (False, 2)  # Video non trovato

    fps = cap.get(cv2.CAP_PROP_FPS) # Ottieni FPS
    total_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))    

    MAX_CONSECUTIVE_ERRORS = int(fps * MAX_GAP_SECONDS) # numero massimo di errori consecutivi

    with mp.solutions.pose.Pose(static_image_mode=True) as pose:    
        for frame_number in tqdm(range(total_frames), desc="Processing frames"):
            try:
                ret, frame = cap.read()
                if not ret:
                    total_frame_error_count += 1
                    consecutive_frame_error_count += 1
                    if total_frame_error_count > total_frames * MAX_TOTAL_ERRORS_PERCENTAGE or consecutive_frame_error_count > MAX_CONSECUTIVE_ERRORS:
                        print("Errore: Troppi frame illegibili")
                        cap.release()
                        return (False, StatusCode.TOO_MANY_MISSING_FRAMES)  # Troppi frame illeggibili
                    continue

                frame = extract_frame_opencv(cap, frame_number)

                if frame is None:
                    print(f"Errore: Frame {frame_number} non trovato in {video_path} (opencv)")
                    frame = extract_frame_ffmpeg(video_path, cap, frame_number)
                    if frame is None:
                        total_frame_error_count += 1
                        consecutive_frame_error_count += 1
                        if total_frame_error_count > total_frames * MAX_TOTAL_ERRORS_PERCENTAGE or consecutive_frame_error_count > MAX_CONSECUTIVE_ERRORS:
                            print("Errore: Troppi frame illegibili")
                            cap.release()
                            return (False, StatusCode.TOO_MANY_MISSING_FRAMES)  # Troppi frame illeggibili
                        continue

                keypoints = extract_keypoints_from_frame(frame, frame_number, fps, pose)

                # Controlla se tutti i valori sono 0
                keypoints_values = [v for k, v in keypoints.items() if k not in ('frame','timestamp','fps')]
                if all(v == 0 for v in keypoints_values):
                    consecutive_frame_error_count += 1
                    if consecutive_frame_error_count > MAX_CONSECUTIVE_ERRORS:
                        print("Errore: Troppi frame illegibili")
                        cap.release()
                        return (False, StatusCode.TOO_MANY_MISSING_FRAMES)  # Troppi frame illeggibili
                    continue

                consecutive_frame_error_count = 0 # resetta il contatore se i keypoints sono validi

                output_image_path = os.path.join(video_frames_dir, f"frame_{frame_number:05d}.jpg")
                success = cv2.imwrite(output_image_path, frame)

                if not success:
                    print(f"Errore: impossibile salvare frame {frame_number} in {output_image_path}")
                    print(f"Tipo frame: {type(frame)}, shape: {None if frame is None else frame.shape}")
                else:
                    print(f"Frame salvato: {output_image_path}")

                buffer.append(keypoints)

                # Scrive ogni BATCH_SIZE righe
                if len(buffer) >= BATCH_SIZE:
                    save_keypoints_list_to_csv(buffer, output_temp_csv)
                    buffer = []  # svuota il buffer

            except Exception as e:
                import traceback
                print("Errore generico durante l'elaborazione del frame:")
                traceback.print_exc()  # stampa il traceback completo
                return (False, StatusCode.GENERIC_ERROR)  # Errore generico

    cap.release()

    # Scrive gli eventuali dati rimanenti
    if buffer:
        save_keypoints_list_to_csv(buffer, output_temp_csv)

    return (True, StatusCode.OK)  # Successo

def save_keypoints_list_to_csv(keypoints_list, output_csv):    

    """
    Salva una lista di keypoints in un file CSV.

    La funzione filtra i keypoints con tutti i valori a 0 (escludendo 'id' e 'class')
    e li salva in un file CSV. Se il file esiste già, carica i dati esistenti e li
    aggiunge ai nuovi dati.

    Args: 
        keypoints_list (list[dict]): Lista di dizionari contenenti le coordinate dei keypoints.
            Ogni dizionario deve contenere le chiavi 'id', 'class' e le coordinate dei keypoints.
        output_csv (str): Percorso del file CSV di output.

    Returns:
        None
    """
    if not keypoints_list:
        return  # Evita di scrivere se la lista è vuota

    file_exists = os.path.isfile(output_csv)

    # Usa i fieldnames dal primo dizionario della lista
    fieldnames = list(keypoints_list[0].keys())

    with open(output_csv, mode='a', newline='') as file:
        writer = csv.DictWriter(file, fieldnames=fieldnames)

        if not file_exists:
            writer.writeheader()

        for keypoints in keypoints_list:
            if is_all_keypoints_zero(keypoints):
                continue  # Salta righe con tutti i keypoints a 0.0

            writer.writerow(keypoints)

def save_keypoints_list_to_json(keypoints_list, output_json):
    """
    Salva una lista di keypoints in un file JSON.

    Args:
        keypoints_list (list[dict]): Lista di dizionari contenenti le coordinate dei keypoints.
            Ogni dizionario deve contenere le chiavi 'id', 'class' e le coordinate dei keypoints.
        output_json (str): Percorso del file JSON di output.

    Returns:
        None
    """
    if not keypoints_list:
        return  # Evita di scrivere se la lista è vuota

    # Filtra i keypoints con tutti i valori a 0 (escludendo 'id' e 'class')
    keypoints_list = [kp for kp in keypoints_list if not is_all_keypoints_zero(kp)]

    if not keypoints_list:
        return  # Evita di salvare se tutti erano nulli

    # Se il file esiste già, carica i dati esistenti
    if os.path.exists(output_json):
        with open(output_json, 'r') as f:
            existing_data = json.load(f)
    else:
        existing_data = []

    # Aggiungi i nuovi dati
    existing_data.extend(keypoints_list)

    # Scrivi tutto nel file
    with open(output_json, 'w') as f:
        json.dump(existing_data, f, indent=4)
