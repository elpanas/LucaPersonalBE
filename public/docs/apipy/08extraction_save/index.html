def save_full_keypoints_to_csv(input_csv, output_csv):
    """
    Salva i keypoints estratti da un file csv in un nuovo file csv.

    Args:
        input_csv (str): Percorso del file csv contenente i keypoints
        output_csv (str): Percorso del nuovo file csv contenente i keypoints

    Returns:
        None
    """

    metadati_temporali = [
        # --- metadati temporali ---
        'frame',
        'timestamp',
        'fps',
        'mean_visibility',
    ]

    # Definizione dei nomi dei keypoints
    keypoint_fields = [
        'nose_x', 'nose_y',
        'neck_x', 'neck_y',
        'RShoulder_x', 'RShoulder_y',
        'RElbow_x', 'RElbow_y',
        'RWrist_x', 'RWrist_y',
        'LShoulder_x', 'LShoulder_y',
        'LElbow_x', 'LElbow_y',
        'LWrist_x', 'LWrist_y',
        'RHip_x', 'RHip_y',
        'RKnee_x', 'RKnee_y',
        'RAnkle_x', 'RAnkle_y',
        'LHip_x', 'LHip_y',
        'LKnee_x', 'LKnee_y',
        'LAnkle_x', 'LAnkle_y',
        'REye_x', 'REye_y',
        'LEye_x', 'LEye_y',
        'REar_x', 'REar_y',
        'LEar_x', 'LEar_y'
    ]

    # Definizione degli angoli da salvare
    angle_fields = [
        "flessione_ginocchio_dx",
        "flessione_ginocchio_sx",
        "flessione_anca_dx",
        "flessione_anca_sx",
        "flessione_gomito_dx",
        "flessione_gomito_sx",
        "abduzione_spalla_dx",
        "abduzione_spalla_sx",
        "inclinazione_collo"
    ]

    ampiezza_fields = [
        'ampiezza_falcata',
        'lunghezza_passo'
    ]

    simmetry_fields = [
        "asimmetria_flessione_ginocchio_dx_sx",
        "asimmetria_flessione_anca_dx_sx",
        "asimmetria_flessione_gomito_dx_sx",
        "asimmetria_abduzione_spalla_dx_sx"
    ]

    fieldnames = (metadati_temporali
                + keypoint_fields 
                + angle_fields 
                + ampiezza_fields 
                + simmetry_fields) # + ["pathology"]

    with open(input_csv, 'r', newline='') as infile, open(output_csv, 'w', newline='') as outfile:
        reader = csv.DictReader(infile)
        writer = csv.DictWriter(outfile, fieldnames=fieldnames)
        writer.writeheader()

        for row in reader:
            keypoints_raw = {k: float(v) if v not in (None, "", "nan") else 0.0 for k, v in row.items()}
            angoli = calcola_angoli(keypoints_raw)
            falcata = calcola_ampiezza_falcata(keypoints_raw)
            passo = calcola_lunghezza_passo(keypoints_raw)

            # Calcolo simmetria riga per riga (valore assoluto differenza dx - sx)
            simmetria = {
                "asimmetria_flessione_ginocchio_dx_sx": calcolo_simmetrie(angoli, "flessione_ginocchio_dx", "flessione_ginocchio_sx"),
                "asimmetria_flessione_anca_dx_sx": calcolo_simmetrie(angoli, "flessione_anca_dx", "flessione_anca_sx"),
                "asimmetria_flessione_gomito_dx_sx": calcolo_simmetrie(angoli, "flessione_gomito_dx", "flessione_gomito_sx"),
                "asimmetria_abduzione_spalla_dx_sx": calcolo_simmetrie(angoli, "abduzione_spalla_dx", "abduzione_spalla_sx"),
            }

            # Composizione riga finale
            output_row = {}

            # Aggiunta metadati se esistono
            for field in metadati_temporali:
                output_row[field] = row.get(field, "")

            output_row.update({field: keypoints_raw.get(field, "") for field in keypoint_fields})
            output_row.update({field: angoli.get(field, "") for field in angle_fields})
            output_row.update({"ampiezza_falcata": falcata, "lunghezza_passo": passo})
            output_row.update(simmetria)
            #output_row.update({"pathology": pathology})
            writer.writerow(output_row)
