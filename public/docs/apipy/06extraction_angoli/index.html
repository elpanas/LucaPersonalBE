def angolo_tra_vettori(A, B, C):    
    """
    Calcola l'angolo tra i vettori AB e BC

    Args: 
        A (array-like): Punto A
        B (array-like): Punto B
        C (array-like): Punto C

    Returns:
        (float): Angolo tra AB e BC in gradi
    """
    AB = np.array(B) - np.array(A)
    CB = np.array(B) - np.array(C)
    cosine_angle = np.dot(AB, CB) / (np.linalg.norm(AB) * np.linalg.norm(CB) + 1e-6)
    angle = np.arccos(np.clip(cosine_angle, -1.0, 1.0))
    return np.degrees(angle)

def calcola_angoli(keypoints_raw):
    """
    Calcola gli angoli di diverse articolazioni del corpo a partire
    dai keypoints estratti da un frame video

    Args:
        keypoints_raw (dict): Dizionario con i keypoints estratti dal frame

    Returns:
        (dict): Dizionario con gli angoli delle articolazioni del corpo
    """
    k = converti_keypoints(keypoints_raw)

    # Angoli solo dove sono disponibili 3 punti
    return {
        # Ginocchia
        "flessione_ginocchio_dx": angolo_tra_vettori(k["RHip"], k["RKnee"], k["RAnkle"]),
        "flessione_ginocchio_sx": angolo_tra_vettori(k["LHip"], k["LKnee"], k["LAnkle"]),

        # Anche
        "flessione_anca_dx": angolo_tra_vettori(k["RShoulder"], k["RHip"], k["RKnee"]),
        "flessione_anca_sx": angolo_tra_vettori(k["LShoulder"], k["LHip"], k["LKnee"]),

        # Gomiti
        "flessione_gomito_dx": angolo_tra_vettori(k["RShoulder"], k["RElbow"], k["RWrist"]),
        "flessione_gomito_sx": angolo_tra_vettori(k["LShoulder"], k["LElbow"], k["LWrist"]),

        # Spalle
        "abduzione_spalla_dx": angolo_tra_vettori(k["RElbow"], k["RShoulder"], k["RHip"]),
        "abduzione_spalla_sx": angolo_tra_vettori(k["LElbow"], k["LShoulder"], k["LHip"]),

        # Collo (stimato come punto medio tra le spalle)
        "inclinazione_collo": angolo_tra_vettori(k["LEar"], k["neck"], k["REar"]),
    }

def converti_keypoints(keypoints):

    """
    Converte un dizionario raw tipo {'LHip_x': ..., 'LHip_y': ...}
    in un dizionario con coordinate (x, y) per punto del corpo

    Args: 
        keypoints (dict): Dizionario con i keypoints estratti dal frame

    Returns: 
        (dict): Dizionario con le coordinate (x, y) per punto del corpo
    """
    convertito = {}
    for k in keypoints:
        if k.endswith('_x'):
            base = k[:-2]
            x = keypoints.get(f"{base}_x", 0)
            y = keypoints.get(f"{base}_y", 0)
            convertito[base] = [x, y]
    return convertito
