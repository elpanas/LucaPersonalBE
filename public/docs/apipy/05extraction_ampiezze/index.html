def calcola_altezza(keypoints):        

    """
    Calcola l'altezza del soggetto rispetto alla media delle caviglie.

    Args: 
        keypoints (dict): Dizionario con i keypoints estratti dal frame

    Returns:    
        (float): Altezza del soggetto rispetto alla media delle caviglie
    """

    nose_y = keypoints.get("nose_y", 0.0)
    lankle_y = keypoints.get("LAnkle_y", 0.0)
    rankle_y = keypoints.get("RAnkle_y", 0.0)
    media_caviglie_y = (lankle_y + rankle_y) / 2
    altezza = abs(nose_y - media_caviglie_y)
    return altezza if altezza != 0 else 1e-6  # per evitare divisioni per 0

def calcola_ampiezza_falcata(keypoints):

    """
    Calcola l'ampiezza falcata del passo rispetto all'altezza del soggetto.

    Args:
        keypoints (dict): Dizionario con i keypoints estratti dal frame

    Returns:
        (float): Ampiezza falcata del passo rispetto all'altezza del soggetto
    """
    x_dx = keypoints.get("RAnkle_x", 0.0)
    y_dx = keypoints.get("RAnkle_y", 0.0)
    x_sx = keypoints.get("LAnkle_x", 0.0)
    y_sx = keypoints.get("LAnkle_y", 0.0)

    altezza = calcola_altezza(keypoints)
    if altezza == 0:
        return 0.0

    ampiezza = math.sqrt((x_dx - x_sx) ** 2 + (y_dx - y_sx) ** 2)
    return ampiezza / altezza

def calcola_lunghezza_passo(keypoints):
    """
    Calcola la lunghezza del passo di un frame video dato il dizionario dei keypoints

    Args:
        keypoints (dict): Dizionario con i keypoints estratti dal frame

    Returns:
        (float): Lunghezza del passo in pixel
    """
    # Coordinate caviglie
    x_dx = keypoints.get("RAnkle_x", 0.0)
    x_sx = keypoints.get("LAnkle_x", 0.0)

    # Altezza stimata come distanza verticale tra naso e media caviglie
    altezza = calcola_altezza(keypoints)
    if altezza == 0:
        return 0.0

    lunghezza = abs(x_dx - x_sx)
    return lunghezza / altezza
