@app.delete("/delete/{uid}/{filename}")
def delete(uid: str, filename: str):    
    """
    Cancella il file CSV associato all'utente con l'UID specificato

    Parameters:
        uid (str): L'UID dell'utente
        filename (str): Il nome del file CSV da cancellare

    Returns:
        (dict): Dizionario contenente lo stato di salute dell'operazione
    """
    file_path = os.path.join(CSV_DIR, uid, filename)
    if os.path.exists(file_path):
        os.remove(file_path)
        return {
            "result": True,
            "code": StatusCode.OK,
            "message": STATUS_MESSAGES.get(StatusCode.OK, "Operazione completata con successo")
        }
    else:
        return {
            "result": False,
            "code": StatusCode.FILE_NOT_FOUND,
            "message": STATUS_MESSAGES.get(StatusCode.FILE_NOT_FOUND, "File non trovato")
        }

@app.get("/downloadcsv/{uid}/{filename}")
def download_csv(uid: str, filename: str):
    """
    Ritorna il file csv richiesto

    Args: 
        uid (str): Identificatore univoco dell'utente
        filename (str): Nome del file csv da scaricare

    Returns: 
        (FileResponse): Il file csv richiesto se esiste, None altrimenti
    """
    file_path = os.path.join(CSV_DIR, uid, filename)

    if not os.path.isfile(file_path):
        return {
            "result": False,
            "code": StatusCode.FILE_NOT_FOUND,
            "message": STATUS_MESSAGES.get(StatusCode.FILE_NOT_FOUND, "File non trovato")
        }

    return FileResponse(
        path=file_path,
        media_type='text/csv',
        filename=filename
    )

@app.get("/extractkeys/{uid}/{filename}")
def extract(uid: str, filename: str):    
    """
    Estrae i keypoints base, derivati e li salva in un file csv.

    Args:
        uid (str): Identificativo dell'utente
        filename (str): Nome del file video da elaborare

    Returns:
        (dict)
            - result:
                - True se l'operazione è andata con successo
                - False altrimenti
            - code:
                - StatusCode.OK se l'operazione è andata con successo
                - StatusCode.GENERIC_ERROR altrimenti
            - message: messaggio di stato dell'operazione
    """
    basename = os.path.splitext(filename)[0] # estrae il nome del file senza estensione
    video_path = os.path.join(VIDEO_DIR, uid, filename) # percorso del video
    user_csv_dir = os.path.join(CSV_DIR, uid) # percorso della cartella con i csv
    video_frames_dir = os.path.join(FRAMES_DIR, uid, basename)
    os.makedirs(user_csv_dir, exist_ok=True)
    os.makedirs(video_frames_dir, exist_ok=True)

    temp_keypoints_filepath = os.path.join(user_csv_dir, f"temp_keypoints_{basename}.csv") # file con i keypoints base
    output_filepath = os.path.join(user_csv_dir, f"keypoints_{basename}.csv") # file con i keypoints base + completi

    try:
        success, code = kp_extraction(video_frames_dir, video_path, temp_keypoints_filepath)

        if not success:
            return {
                "result": False,
                "code": int(code),  # enum -> numero
                "message": STATUS_MESSAGES.get(code, "Errore sconosciuto"),
            }
        else:
            save_full_keypoints_to_csv(temp_keypoints_filepath, output_filepath)

            if os.path.exists(temp_keypoints_filepath): # Elimina il file intermedio
                os.remove(temp_keypoints_filepath)

            return {
                "result": True,
                "code": int(StatusCode.OK), # 0
                "message": STATUS_MESSAGES.get(StatusCode.OK, "Estrazione completata con successo"),
            }

    except Exception as e:
        return {
            "result": False,
            "code": int(StatusCode.GENERIC_ERROR),  # 1
            "message": STATUS_MESSAGES.get(StatusCode.GENERIC_ERROR, "Errore sconosciuto"),
        }

def extract_frames(uid: str, filename: str):

    Estrae i frame dal video e li salva in una cartella

    Parameters
    ----------
    uid : str
        Identificativo dell'utente
    filename : str

        (dict):

                - True se l'operazione è andata con successo
                - False altrimenti
            - code: StatusCode.OK se l'operazione è andata con successo, StatusCode.GENERIC_ERROR altrimenti

@app.get("/videos/{uid}")
def get_videos(uid: str):
    """
    Ritorna la lista dei file video per l'utente specificato.

    Parameters
    ----------
    uid (str): Identificativo dell'utente

    Returns: 
        (dict):
            - result: True se l'operazione è andata con successo, False altrimenti
            - code: 
                - StatusCode.OK se l'operazione è andata con successo
                - StatusCode.VIDEO_NOT_FOUND altrimenti
            - message: messaggio di stato dell'operazione
            - videos: lista dei file video se l'operazione è andata con successo, altrimenti una lista vuota
    """
    user_video_dir = os.path.join(VIDEO_DIR, uid)
    os.makedirs(os.path.dirname(user_video_dir), exist_ok=True)
    videos = os.listdir(user_video_dir)
    if not videos:
        return {
            "result": False,
            "code": StatusCode.VIDEO_NOT_FOUND,
            "message": STATUS_MESSAGES.get(StatusCode.VIDEO_NOT_FOUND, "Video non trovato"),
            "videos": []
        }
    else:
        return {
            "result": True,
            "code": StatusCode.OK,
            "videos": videos
        }
